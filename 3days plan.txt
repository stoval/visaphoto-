```ts name=capacitor.config.ts
import type { CapacitorConfig } from '@capacitor/cli';

const config: CapacitorConfig = {
  appId: 'com.stoval.passportphoto',
  appName: '证件照制作',
  webDir: '.', 
  bundledWebRuntime: false,
  server: {
    androidScheme: 'https'
  }
};

export default config;
```

```bash name=build-and-run.sh
# 1) 安装
npm i @capacitor/core @capacitor/cli
npx cap init "证件照制作" "com.stoval.passportphoto"

# 2) 添加安卓
npx cap add android

# 3) 每次网页代码更新后同步
npx cap sync android

# 4) 用 Android Studio 打开
npx cap open android
```

```bash name=release-aab-notes.sh
# Android Studio -> Build -> Generate Signed Bundle / APK
# 选择 Android App Bundle (AAB)
# 创建/选择 keystore 后生成 app-release.aab
```

最关键两点提醒：
1. `webDir: '.'` 仅适合你现在这种静态单文件/静态站点；如果后面有构建目录（如 `dist`），改成 `dist`。  
2. 你的 `sw.js` 缓存版本要递增（如 `visaphoto-cache-v3`），否则新包里可能还读旧缓存。
